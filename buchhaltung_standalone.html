سیستم لاگ فعال است. از حالا هر فراخوانی Gemini (چه ادمین، چه مشتری، چه موفق، چه ناموفق) توی جدول gemini_usage_log ثبت می‌شه. ۷۲۹ تصویر بالا مربوط به قبل از فعال‌شدن لاگ است و از روی گالری برآورد شده.
کوئری‌های آماده برای دیدن هزینه در Supabase 
-- Gesamtkosten bis heute
SELECT COUNT(*) AS bilder, ROUND(SUM(est_cost_usd),2) AS usd
FROM gemini_usage_log WHERE success;

-- Kosten pro Monat
SELECT to_char(created_at,'YYYY-MM') AS monat,
       COUNT(*) AS bilder, ROUND(SUM(est_cost_usd),2) AS usd
FROM gemini_usage_log WHERE success GROUP BY 1 ORDER BY 1 DESC;

-- Admin vs. Kunde
SELECT is_admin, COUNT(*) AS bilder, ROUND(SUM(est_cost_usd),2) AS usd
FROM gemini_usage_log WHERE success GROUP BY 1;
